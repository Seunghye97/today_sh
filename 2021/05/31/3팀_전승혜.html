store = {}

while True:
    product = input('물건이름: ')
    if product=='' :
        break
    num = int(input('개수: '))
    if product in store.keys():
        store[product] += num  #숫자 더하기
    else:
        store[product] = num
print(store)

#오름차순 정렬

    print(key, value, end = ' ')

print(sweety.union(salty))
print(sweety|salty)

print(sweety.intersection(salty))
print(sweety&salty)
#차집합
print(sweety.difference(salty))
print(sweety-salty)
print(salty.difference(sweety))
print(salty-sweety)
